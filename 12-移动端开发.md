## 十二、移动端开发

### 12.1 移动端基础

#### 12.1.1 视口与适配

```html
<!-- 视口 meta 标签 -->
<meta name="viewport" content="width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=no">
```

**适配方案：**

```css
/* 1. rem 适配 */
/* 设置根字体大小 */
html {
    font-size: 16px;
}

/* 根据屏幕宽度动态设置 */
@media (max-width: 375px) {
    html { font-size: 14px; }
}
@media (min-width: 750px) {
    html { font-size: 20px; }
}

/* 使用 rem */
.container {
    width: 10rem;  /* 160px */
    font-size: 1rem;  /* 16px */
}

/* 2. vw/vh 适配 */
.container {
    width: 100vw;
    height: 100vh;
    font-size: 4vw;  /* 视口宽度的 4% */
}

/* 3. flexible.js 方案（阿里） */
/* 将视口分为 10 份，1rem = 视口宽度/10 */

/* 4. postcss-px-to-viewport */
/* 自动将 px 转换为 vw */
```

**1px 边框问题：**

```css
/* 问题：高清屏（DPR=2）上 1px 显示为 2px */

/* 方案1：transform scale */
.border-1px {
    position: relative;
}
.border-1px::after {
    content: '';
    position: absolute;
    left: 0;
    bottom: 0;
    width: 100%;
    height: 1px;
    background: #ccc;
    transform: scaleY(0.5);
    transform-origin: 0 0;
}

/* 方案2：viewport + rem */
/* 设置 initial-scale=0.5，然后所有尺寸 ×2 */

/* 方案3：box-shadow */
.border-1px {
    box-shadow: 0 1px 0 0 #ccc;
}

/* 方案4：媒体查询 */
@media (-webkit-min-device-pixel-ratio: 2) {
    .border-1px {
        border-width: 0.5px;
    }
}
```

**300ms 点击延迟：**

```html
<!-- 方案1：禁用缩放 -->
<meta name="viewport" content="width=device-width, user-scalable=no">

<!-- 方案2：touch-action -->
<style>
    * { touch-action: manipulation; }
</style>

<!-- 方案3：FastClick 库 -->
<script>
    FastClick.attach(document.body);
</script>
```

### 12.2 微信小程序

```javascript
// app.js
App({
    onLaunch(options) {
        console.log('小程序启动');
    },
    onShow(options) {
        console.log('小程序显示');
    },
    onHide() {
        console.log('小程序隐藏');
    },
    globalData: {
        userInfo: null
    }
});

// pages/index/index.js
Page({
    data: {
        message: 'Hello',
        list: []
    },
    
    onLoad(options) {
        console.log('页面加载', options);
        this.fetchData();
    },
    
    onShow() {
        console.log('页面显示');
    },
    
    onReady() {
        console.log('页面初次渲染完成');
    },
    
    onHide() {
        console.log('页面隐藏');
    },
    
    onUnload() {
        console.log('页面卸载');
    },
    
    async fetchData() {
        const res = await wx.request({
            url: 'https://api.example.com/data',
            method: 'GET'
        });
        this.setData({ list: res.data });
    },
    
    handleTap(e) {
        const { id } = e.currentTarget.dataset;
        wx.navigateTo({
            url: `/pages/detail/detail?id=${id}`
        });
    }
});
```

```xml
<!-- pages/index/index.wxml -->
<view class="container">
    <text>{{message}}</text>
    
    <view wx:for="{{list}}" wx:key="id">
        <text>{{item.name}}</text>
        <button bindtap="handleTap" data-id="{{item.id}}">查看详情</button>
    </view>
    
    <input bindinput="handleInput" placeholder="请输入" />
    
    <image src="{{imageUrl}}" mode="aspectFill" />
</view>
```
