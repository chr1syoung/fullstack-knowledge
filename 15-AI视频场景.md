## 11.x.x Canvas 绘制视频进度条预览图

#### 知识点详解

**使用 Canvas 绘制视频预览图：**

```javascript
// 获取视频元素
const video = document.getElementById('video');
const canvas = document.getElementById('previewCanvas');
const ctx = canvas.getContext('2d');

// 设置 Canvas 尺寸
canvas.width = 160;   // 预览图宽度
canvas.height = 90;  // 16:9 比例

// 绘制当前帧到 Canvas
function drawFrame() {
    ctx.drawImage(video, 0, 0, canvas.width, canvas.height);
}

// 视频加载完成后绘制第一帧
video.addEventListener('loadedmetadata', () => {
    video.currentTime = 0;  // 跳到开头
});

video.addEventListener('seeked', () => {
    drawFrame();  // 跳转到指定时间后绘制
});
```

**生成缩略图条（进度条预览）：**

```javascript
// 生成多帧预览图
async function generateVideoThumbnail(videoUrl, thumbCount = 10) {
    const video = document.createElement('video');
    const canvas = document.createElement('canvas');
    const ctx = canvas.getContext('2d');
    
    // 预览图尺寸
    const thumbWidth = 80;
    const thumbHeight = 45;
    canvas.width = thumbWidth;
    canvas.height = thumbHeight;
    
    video.src = videoUrl;
    video.muted = true;
    
    // 等待视频加载
    await new Promise(resolve => {
        video.onloadedmetadata = resolve;
    });
    
    const duration = video.duration;
    const thumbnails = [];
    
    // 均匀采样 thumbCount 个帧
    for (let i = 0; i < thumbCount; i++) {
        const time = (duration / thumbCount) * i;
        video.currentTime = time;
        
        await new Promise(resolve => {
            video.onseeked = resolve;
        });
        
        // 绘制到 Canvas
        ctx.drawImage(video, 0, 0, thumbWidth, thumbHeight);
        
        // 转为 DataURL
        const dataUrl = canvas.toDataURL('image/jpeg', 0.6);
        thumbnails.push({
            time,
            dataUrl
        });
    }
    
    return thumbnails;
}

// 使用
generateVideoThumbnail('/path/to/video.mp4').then(thumbs => {
    console.log('生成了', thumbs.length, '张预览图');
    // thumbs[0].time  = 0s
    // thumbs[0].dataUrl = "data:image/jpeg;base64,..."
});
```

```javascript
const progressBar = document.querySelector('.progress-bar');
const previewContainer = document.querySelector('.preview-container');

progressBar.addEventListener('mousemove', async (e) => {
    const rect = progressBar.getBoundingClientRect();
```

```javascript
// 绘制圆角矩形（预览框）
function drawRoundedRect(ctx, x, y, width, height, radius) {
    ctx.beginPath();
    ctx.moveTo(x + radius, y);
    ctx.lineTo(x + width - radius, y);
    ctx.arcTo(x + width, y, x + width, y + radius, radius);
    ctx.lineTo(x + width, y + height - radius);
    ctx.arcTo(x + width, y + height, x + width - radius, y + height, radius);
    ctx.lineTo(x + radius, y + height);
    ctx.arcTo(x, y + height, x, y + height - radius, radius);
    ctx.lineTo(x, y + radius);
    ctx.arcTo(x, y, x + radius, y, radius);
    ctx.closePath();
}
```

**流式渲染核心思路：**

```javascript
// 使用 SSE 或 Fetch Stream 接收数据
async function streamChat(messages) {
    const response = await fetch('/api/chat/stream', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ messages })
    });
    
    const reader = response.body.getReader();
    const decoder = new TextDecoder();
    let content = '';
    
    while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        
        const chunk = decoder.decode(value);
        content += chunk;
        
        // 实时更新 UI
        updateMessageContent(content);
    }
}
```

**Markdown 解析与渲染：**

```javascript
// 使用 marked 或 react-markdown
import { marked } from 'marked';
import DOMPurify from 'dompurify';

// 实时解析 Markdown
function renderMarkdownIncremental(text) {
    // 增量解析：只解析新增部分
    const lines = text.split('\n');
    const html = lines.map(line => marked.parseInline(line)).join('<br>');
    
    // 防止 XSS
    return DOMPurify.sanitize(html);
}

// React 组件实现
function AIMessage({ content, isStreaming }) {
    const [displayContent, setDisplayContent] = useState('');
    
    useEffect(() => {
        if (!isStreaming) {
            setDisplayContent(content);
            return;
        }
        
        // 流式更新（简单的节流）
        let index = 0;
        const timer = setInterval(() => {
            index++;
            setDisplayContent(content.slice(0, index));
            if (index >= content.length) clearInterval(timer);
        }, 20);
        
        return () => clearInterval(timer);
    }, [content, isStreaming]);
    
    // 解析 Markdown
    const html = marked.parse(displayContent);
    
    return (
        <div 
            className="ai-message"
            dangerouslySetInnerHTML={{ __html: DOMPurify.sanitize(html) }}
        />
    );
}
```

```javascript
// 1. 代码块流式渲染（避免闪烁）
function CodeBlockRenderer({ code, language }) {
    const [lines, setLines] = useState([]);
    
    useEffect(() => {
        // 按行增量添加
        const newLines = code.split('\n');
        newLines.forEach((line, i) => {
            setTimeout(() => {
                setLines(prev => [...prev, line]);
            }, i * 10);
        });
    }, [code]);
    
    return (
        <pre>
            <code>
                {lines.map((line, i) => (
                    <span key={i}>
                        {line}
                        {i < lines.length - 1 && '\n'}
                    </span>
                ))}
            </code>
        </pre>
    );
}

// 2. 打字机效果
function Typewriter({ text, speed = 30 }) {
    const [display, setDisplay] = useState('');
    
    useEffect(() => {
        let i = 0;
        const timer = setInterval(() => {
            setDisplay(text.slice(0, i + 1));
            i++;
            if (i >= text.length) clearInterval(timer);
        }, speed);
        return () => clearInterval(timer);
    }, [text, speed]);
    
    return <span>{display}</span>;
}
```

```javascript
function fillTemplate(templateName, values) {
    const template = promptTemplates[templateName];
```

```javascript
// Prompt 历史管理
class PromptHistory {
    constructor(maxSize = 100) {
        this.history = [];
        this.maxSize = maxSize;
    }
    
    add(prompt, params, result) {
        this.history.unshift({
            prompt,
            params,
            result,
            timestamp: Date.now()
        });
        
        if (this.history.length > this.maxSize) {
            this.history.pop();
        }
    }
    
    search(keyword) {
        return this.history.filter(h => 
            h.prompt.toLowerCase().includes(keyword.toLowerCase())
        );
    }
    
    getRecent(n = 10) {
        return this.history.slice(0, n);
    }
}
```

## 11.x.x AI 接口超时处理

#### 知识点详解

**超时处理策略：**

```javascript
// 1. AbortController 超时控制
async function fetchWithTimeout(url, options = {}, timeout = 30000) {
    const controller = new AbortController();
    const id = setTimeout(() => controller.abort(), timeout);
    
    try {
        const response = await fetch(url, {
            ...options,
            signal: controller.signal
        });
        clearTimeout(id);
        return response;
    } catch (error) {
        clearTimeout(id);
        if (error.name === 'AbortError') {
            throw new Error('请求超时');
        }
        throw error;
    }
}

// 2. 重试机制
async function fetchWithRetry(url, options = {}, maxRetries = 3) {
    for (let i = 0; i < maxRetries; i++) {
        try {
            return await fetch(url, options);
        } catch (error) {
            if (i === maxRetries - 1) throw error;
            await new Promise(r => setTimeout(r, 1000 * (i + 1))); // 指数退避
        }
    }
}

// 3. AI 流式响应超时
async function streamAIResponse(messages, onChunk, options = {}) {
    const { timeout = 60000, retries = 2 } = options;
    
    for (let attempt = 0; attempt <= retries; attempt++) {
        try {
            const controller = new AbortController();
            const timeoutId = setTimeout(() => controller.abort(), timeout);
            
            const response = await fetch('/api/ai/chat', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ messages }),
                signal: controller.signal
            });
            
            clearTimeout(timeoutId);
            
            const reader = response.body.getReader();
            const decoder = new TextDecoder();
            
            while (true) {
                const { done, value } = await reader.read();
                if (done) break;
                
                const chunk = decoder.decode(value);
                onChunk(chunk);
            }
            
            return; // 成功
            
        } catch (error) {
            if (attempt === retries) {
                throw new Error(`AI 响应失败，已重试 ${retries} 次`);
            }
            await new Promise(r => setTimeout(r, 1000));
        }
    }
}
```

**UI 层面的超时体验：**

```javascript
// UI 状态管理
function AIChat() {
    const [status, setStatus] = useState('idle'); // idle, loading, streaming, error
    const [error, setError] = useState(null);
    const [progress, setProgress] = useState(0);
    
    const sendMessage = async (content) => {
        setStatus('loading');
        setProgress(0);
        
        try {
            setStatus('streaming');
            
            await streamAIResponse(
                [{ role: 'user', content }],
                (chunk) => {
                    // 更新内容
                    setProgress(p => Math.min(p + 10, 90));
                },
                {
                    timeout: 60000,

                        setStatus('timeout');
                        setError('AI 响应超时，请稍后重试');

            setStatus('idle');

            setStatus('error');
            setError(error.message);
        }
    };

                <div className="error">
                    {error}

            )}
        </div>
    );
}
```
