**1. 实现 call、apply、bind：**

```javascript
// call
Function.prototype.myCall = function(context, ...args) {
    context = context || window;
    const fn = Symbol('fn');
    context[fn] = this;
    const result = context[fn](...args);
    delete context[fn];
    return result;
};

// apply
Function.prototype.myApply = function(context, args = []) {
    context = context || window;
    const fn = Symbol('fn');
    context[fn] = this;
    const result = context[fn](...args);
    delete context[fn];
    return result;
};

// bind
Function.prototype.myBind = function(context, ...args) {
    const fn = this;
    return function(...moreArgs) {
        return fn.apply(context, [...args, ...moreArgs]);
    };
};
```

**2. 实现 new 操作符：**

```javascript
function myNew(Constructor, ...args) {
    // 1. 创建新对象，原型指向构造函数的 prototype
    const obj = Object.create(Constructor.prototype);
    
    // 2. 执行构造函数
    const result = Constructor.apply(obj, args);
    
    // 3. 如果构造函数返回对象，则返回该对象；否则返回新对象
    return result instanceof Object ? result : obj;
}
```

**3. 实现 Promise：**

```javascript
class MyPromise {
    constructor(executor) {
        this.state = 'pending';
        this.value = undefined;
        this.reason = undefined;
        this.onFulfilledCallbacks = [];
        this.onRejectedCallbacks = [];
        
        const resolve = (value) => {
            if (this.state === 'pending') {
                this.state = 'fulfilled';
                this.value = value;
                this.onFulfilledCallbacks.forEach(fn => fn(value));
            }
        };
        
        const reject = (reason) => {
            if (this.state === 'pending') {
                this.state = 'rejected';
                this.reason = reason;
                this.onRejectedCallbacks.forEach(fn => fn(reason));
            }
        };
        
        try {
            executor(resolve, reject);
        } catch (error) {
            reject(error);
        }
    }
    
    then(onFulfilled, onRejected) {
        onFulfilled = typeof onFulfilled === 'function' ? onFulfilled : v => v;
        onRejected = typeof onRejected === 'function' ? onRejected : r => { throw r; };
        
        return new MyPromise((resolve, reject) => {
            const handleFulfilled = (value) => {
                try {
                    const result = onFulfilled(value);
                    if (result instanceof MyPromise) {
                        result.then(resolve, reject);
                    } else {
                        resolve(result);
                    }
                } catch (error) {
                    reject(error);
                }
            };
            
            const handleRejected = (reason) => {
                try {
                    const result = onRejected(reason);
                    if (result instanceof MyPromise) {
                        result.then(resolve, reject);
                    } else {
                        resolve(result);
                    }
                } catch (error) {
                    reject(error);
                }
            };
            
            if (this.state === 'fulfilled') {
                setTimeout(() => handleFulfilled(this.value));
            } else if (this.state === 'rejected') {
                setTimeout(() => handleRejected(this.reason));
            } else {
                this.onFulfilledCallbacks.push(handleFulfilled);
                this.onRejectedCallbacks.push(handleRejected);
            }
        });
    }
    
    catch(onRejected) {
        return this.then(null, onRejected);
    }
    
    finally(onFinally) {
        return this.then(
            value => MyPromise.resolve(onFinally()).then(() => value),
            reason => MyPromise.resolve(onFinally()).then(() => { throw reason; })
        );
    }
    
    static resolve(value) {
        if (value instanceof MyPromise) return value;
        return new MyPromise(resolve => resolve(value));
    }
    
    static reject(reason) {
        return new MyPromise((_, reject) => reject(reason));
    }
    
    static all(promises) {
        return new MyPromise((resolve, reject) => {
            const results = [];
            let count = 0;
            
            promises.forEach((promise, index) => {
                MyPromise.resolve(promise).then(value => {
                    results[index] = value;
                    if (++count === promises.length) {
                        resolve(results);
                    }
                }, reject);
            });
        });
    }
}
```

**5. 柯里化实现：**

```javascript
function curry(fn) {
    return function curried(...args) {
        if (args.length >= fn.length) {
            return fn.apply(this, args);
        }
        return function(...moreArgs) {
            return curried.apply(this, args.concat(moreArgs));
        };
    };
}

// 测试
const add = (a, b, c) => a + b + c;
const curriedAdd = curry(add);

console.log(curriedAdd(1)(2)(3));  // 6
console.log(curriedAdd(1, 2)(3));  // 6
console.log(curriedAdd(1)(2, 3));  // 6
```

### 11.x.x Canvas 实现电影院选票功能 [重要]

#### 知识点详解

用 Canvas 实现电影院座位图交互，包括座位绘制、点击选中、座位状态渲染等核心功能：

**1. 基础座位图绘制：**

```javascript
class CinemaSeats {
  constructor(canvasId, options = {}) {
    this.canvas = document.getElementById(canvasId);
    this.ctx = this.canvas.getContext('2d');
    this.rows = options.rows || 10;      // 行数
    this.seatsPerRow = options.seats || 20; // 每行座位数
    this.seatWidth = options.seatWidth || 40;
    this.seatHeight = options.seatHeight || 35;
    this.gap = options.gap || 6;
    this.padding = options.padding || 30;

    // 座位状态：available | selected | occupied
    this.seats = this.initSeats();
    this.selectedSeats = new Set();
    this.price = options.price || 60;

    this.canvas.width = this.padding * 2 + this.seatsPerRow * (this.seatWidth + this.gap);
    this.canvas.height = this.padding * 2 + this.rows * (this.seatHeight + this.gap) + 60;

    this.bindEvents();
    this.draw();
  }

  initSeats() {
    const seats = [];
    for (let r = 0; r < this.rows; r++) {
      for (let c = 0; c < this.seatsPerRow; c++) {
        // 随机设置部分座位已被占用
        const state = Math.random() < 0.15 ? 'occupied' : 'available';
        seats.push({ row: r, col: c, state });
      }
    }
    return seats;
  }

  bindEvents() {
    this.canvas.addEventListener('click', (e) => this.handleClick(e));
    this.canvas.addEventListener('mousemove', (e) => this.handleHover(e));
  }

  handleClick(e) {
    const seat = this.getSeatAtPoint(e.offsetX, e.offsetY);
    if (!seat || seat.state === 'occupied') return;

    const key = `${seat.row}-${seat.col}`;
    if (this.selectedSeats.has(key)) {
      this.selectedSeats.delete(key);
      seat.state = 'available';
    } else {
      this.selectedSeats.add(key);
      seat.state = 'selected';
    }
    this.draw();
    this.onSelectionChange?.(Array.from(this.selectedSeats), this.selectedSeats.size * this.price);
  }

  getSeatAtPoint(x, y) {
    const { padding, seatWidth, seatHeight, gap, seatsPerRow } = this;
    const row = Math.floor((y - padding) / (seatHeight + gap));
    const col = Math.floor((x - padding) / (seatWidth + gap));
    if (row < 0 || row >= this.rows || col < 0 || col >= seatsPerRow) return null;
    return this.seats[row * seatsPerRow + col];
  }

  draw() {
    const { ctx, padding, seatWidth, seatHeight, gap } = this;
    ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);

    // 绘制屏幕
    ctx.fillStyle = '#e0e0e0';
    ctx.fillRect(padding, 10, this.seatsPerRow * (seatWidth + gap) - gap, 15);
    ctx.fillStyle = '#999';
    ctx.font = '14px Arial';
    ctx.textAlign = 'center';
    ctx.fillText('■ 屏幕 ■', this.canvas.width / 2, 22);

    // 绘制座位
    this.seats.forEach((seat, i) => {
      const x = padding + seat.col * (seatWidth + gap);
      const y = padding + 40 + seat.row * (seatHeight + gap);

      const colors = {
        available: '#4CAF50',
        selected: '#FF5722',
        occupied: '#9E9E9E'
      };
      const emoji = { available: '○', selected: '●', occupied: '✕' };

      ctx.fillStyle = colors[seat.state];
      ctx.fillRect(x, y, seatWidth, seatHeight);

      ctx.fillStyle = '#fff';
      ctx.font = '16px Arial';
      ctx.textAlign = 'center';
      ctx.textBaseline = 'middle';
      ctx.fillText(emoji[seat.state], x + seatWidth / 2, y + seatHeight / 2);

      // 行号
      if (seat.col === 0) {
        ctx.fillStyle = '#666';
        ctx.font = '12px Arial';
        ctx.textAlign = 'right';
        ctx.fillText(seat.row + 1 + '排', x - 5, y + seatHeight / 2);
      }
    });

    // 绘制已选座位汇总
    ctx.fillStyle = '#333';
    ctx.font = '14px Arial';
    ctx.textAlign = 'left';
    ctx.fillText(`已选 ${this.selectedSeats.size} 个座位，共 ¥${this.selectedSeats.size * this.price}`, padding, this.canvas.height - 15);
  }
}

// 使用
const cinema = new CinemaSeats('canvas', {
  rows: 8,
  seats: 15,
  price: 50
});
```

### 11.x.x 小程序为什么会有两个线程 [重要]

#### 知识点详解

小程序的双线程架构是其核心设计，用于解决 Web 标准的性能和安全问题。

**为什么需要双线程？**

| 问题 | 描述 |
|------|------|
| 性能 | Web 的渲染线程和 JS 线程共用，执行长任务会阻塞渲染 |
| 安全 | JS 可以随意操作 DOM、调用任何 API，无法管控 |
| 管控 | 微信希望统一管控小程序能力，防止滥用 |

**双线程模型：**

```
┌──────────────────────────────────────────────┐
│                   视图层（View）               │
│  ┌────────────────┐  ┌───────────────────┐  │
│  │ WebView（渲染UI）│  │ WXML/WXSS 解析器  │  │
│  └────────────────┘  └───────────────────┘  │
│           ↑ Native 层通信（WeixinJSBridge）     │
│  ┌────────────────────────────────────────┐  │
│  │              逻辑层（App Service）      │  │
│  │  ┌──────────────────────────────────┐  │  │
│  │  │  JS 引擎（V8 / JSCore）运行逻辑  │  │  │
│  │  │  处理 Page/App/Component 定义    │  │  │
│  │  └──────────────────────────────────┘  │  │
│  └────────────────────────────────────────┘  │
└──────────────────────────────────────────────┘
```

**线程职责划分：**

| 线程 | 职责 | 运行内容 |
|------|------|---------|
| **视图层** | 界面渲染 | WebView、WXML、WXSS |
| **逻辑层** | 业务逻辑 | JS 引擎、数据处理、API 调用 |

**通信机制：**

```javascript
// 逻辑层 → 视图层：通过 setData 触发渲染
this.setData({ name: 'Tom' }); // 逻辑层更新数据

// 视图层 → 逻辑层：通过事件触发回调
// <view bindtap="onTap"></view>
// onTap() 在逻辑层执行
```

**与 Web/SPA 对比：**

| 特性 | 小程序 | Web SPA |
|------|--------|--------|
| 线程 | 双线程（逻辑+渲染分离） | 单线程（逻辑+渲染共用） |
| 渲染 | 虚拟 DOM + 直接操作 | 虚拟 DOM diff 后更新 |
| API | 必须通过桥接调用 | 直接调用 |
| 性能 | setData 按需更新 | 可能过度渲染 |
| 管控 | 能力受限制 | 完全自由 |

**数据传递限制：**

```javascript
// ⚠️ 每次 setData 传递有大小限制（默认约 1MB）
// 频繁调用会影响性能
this.setData({
  // 正确：只传变更数据
  'list[0].name': 'New Name',  // 路径写法，减少数据传输
});

// ❌ 错误：每次传整个列表
this.setData({ list: this.data.list }); // 数据量大，性能差
```

**常见面试题：**

> **Q：为什么小程序 setData 是异步的？**
> 因为视图层和逻辑层是分离的，数据需要跨线程传递（逻辑层 → Native → 视图层），所以是异步的。

> **Q：双线程相比单线程的优势？**
> 1. 逻辑层崩溃不会影响视图层（更稳定）
> 2. 微信可以管控 API 调用权限（更安全）
> 3. 逻辑层长任务不阻塞渲染（性能更好）

### 11.x.x SPA 为什么提供 hash 路由，好处是什么 [重要]

#### 知识点详解

SPA 应用中 hash 路由（#xxx）相比 history 路由（/xxx）有独特的优势，是许多框架默认或备选方案。

**为什么 hash 能实现无刷新切换？**

```
完整 URL: https://example.com/page1#/user/123

┌─────────────────┬──────────────┐
│  协议+域名+路径   │   Hash 部分  │
│  (需服务器请求)   │  (客户端处理) │
└─────────────────┴──────────────┘
```

浏览器规则：**改变 hash 不会触发页面刷新，不会向服务器发请求。**

**hash 路由 vs history 路由对比：**

| 特性 | hash 路由（`#/user/123`） | history 路由（`/user/123`） |
|------|-------------------------|--------------------------|
| URL 美观度 | ❌ 带有 `#`，不美观 | ✅ 干净美观 |
| 服务器配置 | ❌ 不需要（前端完全处理） | ⚠️ 需要配置（404 回退） |
| 刷新丢失 | ❌ 不丢失（#在本地） | ⚠️ 刷新后 404（需服务端配置） |
| 微信分享 | ✅ 可携带 # 后参数 | ❌ 需配合 URL 参数 |
| 兼容性 | ✅ 所有浏览器 | ⚠️ 需要 history.pushState API |
| 前进后退 | ✅ 支持 | ✅ 支持（popstate 事件） |
| SEO | ❌ 差（搜索引擎一般不索引 # 后内容） | ✅ 好 |

**History API：**

```javascript
// 切换路由，不刷新页面
history.pushState({ id: 1 }, 'Page 1', '/page1');
history.replaceState({ id: 1 }, 'Page 1', '/page1');

// 监听前进/后退
window.addEventListener('popstate', (e) => {
  console.log('路由变化', e.state);
  // 注意：pushState/replaceState 不会触发 popstate
});

// 获取当前路由
console.log(location.pathname); // history 模式
console.log(location.hash);    // hash 模式
```

**服务端配置（history 模式必须）：**

```nginx
# Nginx 配置
location / {
  try_files $uri $uri/ /index.html;
}

# 或者 Apache .htaccess
RewriteEngine On
RewriteBase /
RewriteCond %{REQUEST_FILENAME} !-f
RewriteCond %{REQUEST_FILENAME} !-d
RewriteRule . /index.html [L]
```

**为什么 hash 模式不需要服务器配置？**

因为无论如何刷新，服务器只看到 `#` 前面的部分（`https://example.com`），服务器永远返回 `index.html`，# 后面的内容由前端 JS 处理。

**hash 路由的典型应用：**

| 场景 | 说明 |
|------|------|
| 旧项目迁移 | 不想改服务器配置，直接用 hash |
| 微信内嵌 H5 | 分享页面使用 hash 路由更稳定 |
| GitHub Pages | 静态托管无法配置服务端，必须用 hash 或 404 回退 |

**常见面试题：**

> **Q：hash 路由和 history 路由各适合什么场景？**
> - hash：微信 H5、分享页面、不想配置服务器、兼容老浏览器
> - history：正常 Web 应用，追求 URL 美观、SEO 要求高

> **Q：hash 路由会影响 SEO 吗？**
> 是的，搜索引擎通常不会抓取 `#` 后面的内容。如果对 SEO 有要求，应使用 history 路由。

## 14.x WebGPU：AI 视频前端的下一个变革

#### 知识点详解

**WebGPU 是什么？**

WebGPU 是新一代 Web 图形 API，是 WebGL 的继任者，提供对 GPU 的更底层访问能力。

```
WebGL（旧）：基于 OpenGL ES，API 设计老旧，CPU 开销大
WebGPU（新）：基于 Vulkan/Metal/D3D12，现代化设计，支持计算着色器
```

**WebGPU vs WebGL：**

| 特性 | WebGL | WebGPU |
|------|-------|--------|
| 底层 API | OpenGL ES | Vulkan/Metal/D3D12 |
| 计算着色器 | ❌ | ✅ |
| GPU 计算（非图形） | 有限 | 完整支持 |
| 多线程 | 有限 | 支持 |
| 性能 | 较好 | 更好 |
| 浏览器支持 | 全面 | Chrome 113+（逐步普及） |

**WebGPU 基础用法：**

```javascript
// 1. 初始化
const adapter = await navigator.gpu.requestAdapter();
const device = await adapter.requestDevice();

// 2. 创建 Canvas 上下文
const canvas = document.getElementById('canvas');
const context = canvas.getContext('webgpu');
const format = navigator.gpu.getPreferredCanvasFormat();
context.configure({ device, format });

// 3. 计算着色器（WGSL 语言）
const shaderCode = `
@group(0) @binding(0) var<storage, read> input: array<f32>;
@group(0) @binding(1) var<storage, read_write> output: array<f32>;

@compute @workgroup_size(64)
fn main(@builtin(global_invocation_id) id: vec3<u32>) {
    let i = id.x;
    // 图像处理：灰度转换
    output[i] = input[i * 4] * 0.299 +
                input[i * 4 + 1] * 0.587 +
                input[i * 4 + 2] * 0.114;
}
`;

// 4. 创建计算管线
const shaderModule = device.createShaderModule({ code: shaderCode });
const pipeline = device.createComputePipeline({
    layout: 'auto',
    compute: { module: shaderModule, entryPoint: 'main' }
});

// 5. 执行 GPU 计算
async function processImageOnGPU(imageData) {
    // 创建 GPU Buffer
    const inputBuffer = device.createBuffer({
        size: imageData.data.byteLength,
        usage: GPUBufferUsage.STORAGE | GPUBufferUsage.COPY_DST
    });

    const outputBuffer = device.createBuffer({
        size: imageData.data.byteLength / 4,
        usage: GPUBufferUsage.STORAGE | GPUBufferUsage.COPY_SRC
    });

    // 上传数据到 GPU
    device.queue.writeBuffer(inputBuffer, 0, imageData.data);

    // 绑定组
    const bindGroup = device.createBindGroup({
        layout: pipeline.getBindGroupLayout(0),
        entries: [
            { binding: 0, resource: { buffer: inputBuffer } },
            { binding: 1, resource: { buffer: outputBuffer } }
        ]
    });

    // 提交计算命令
    const encoder = device.createCommandEncoder();
    const pass = encoder.beginComputePass();
    pass.setPipeline(pipeline);
    pass.setBindGroup(0, bindGroup);
    pass.dispatchWorkgroups(Math.ceil(imageData.width * imageData.height / 64));
    pass.end();

    device.queue.submit([encoder.finish()]);

    // 读取结果
    const readBuffer = device.createBuffer({
        size: outputBuffer.size,
        usage: GPUBufferUsage.COPY_DST | GPUBufferUsage.MAP_READ
    });

    const readEncoder = device.createCommandEncoder();
    readEncoder.copyBufferToBuffer(outputBuffer, 0, readBuffer, 0, outputBuffer.size);
    device.queue.submit([readEncoder.finish()]);

    await readBuffer.mapAsync(GPUMapMode.READ);
    return new Float32Array(readBuffer.getMappedRange());
}
```

```javascript
    init() {
```

```javascript
class PerformanceMonitor {
    collectWebVitals() {
        // LCP（最大内容绘制）
        new PerformanceObserver((list) => {
            const entries = list.getEntries();
            const lcp = entries[entries.length - 1];
            this.report({ type: 'LCP', value: lcp.startTime });
        }).observe({ entryTypes: ['largest-contentful-paint'] });

        // FID（首次输入延迟）
        new PerformanceObserver((list) => {
            list.getEntries().forEach(entry => {
                this.report({
                    type: 'FID',
                    value: entry.processingStart - entry.startTime
                });
            });
        }).observe({ entryTypes: ['first-input'] });

        let clsValue = 0;

            list.getEntries().forEach(entry => {
                if (!entry.hadRecentInput) {
                    clsValue += entry.value;
                }
            });
            this.report({ type: 'CLS', value: clsValue });
        }).observe({ entryTypes: ['layout-shift'] });
```

**数据上报策略：**

```javascript
class Reporter {
    constructor() {
        this.queue = [];
        this.maxQueueSize = 20;
        this.flushInterval = 5000;  // 5秒批量上报

        // 定时批量上报
        setInterval(() => this.flush(), this.flushInterval);

        // 页面关闭前上报
        window.addEventListener('beforeunload', () => this.flush());
        document.addEventListener('visibilitychange', () => {
            if (document.visibilityState === 'hidden') this.flush();
        });
    }

    add(data) {
        this.queue.push(data);
        // 错误立即上报
        if (data.type?.includes('error')) {
            this.flush();
        } else if (this.queue.length >= this.maxQueueSize) {
            this.flush();
        }
    }

    flush() {
        if (this.queue.length === 0) return;
        const data = [...this.queue];
        this.queue = [];

        // sendBeacon 不受页面关闭影响
        navigator.sendBeacon('/api/monitor/batch', JSON.stringify(data));
    }
}
```

```html
document.getElementById('subapp').contentWindow.postMessage(
    { type: 'TOKEN', data: 'xxx' },
    'http://sub-app.com'
);
</script>
```

**方案对比：**

| 方案 | 隔离性 | 通信 | 性能 | 适用场景 |
|------|--------|------|------|---------|
| qiankun | JS沙箱+CSS隔离 | props/事件总线 | 好 | 主流选择 |
| Module Federation | 无隔离 | 直接共享 | 最好 | 同技术栈 |
| iframe | 最彻底 | postMessage | 差 | 强隔离需求 |

**微前端通信方案：**

```javascript
// 方案1：props 传递（父→子）
// 方案2：全局状态（qiankun initGlobalState）
import { initGlobalState } from 'qiankun';

const state = initGlobalState({ user: null, token: '' });

// 主应用修改
state.setGlobalState({ token: 'new-token' });

// 子应用监听
props.onGlobalStateChange((state, prev) => {
    console.log('状态变化:', state);
});

// 方案3：自定义事件总线
const eventBus = {
    events: {},
    on(event, cb) { (this.events[event] = this.events[event] || []).push(cb); },
    emit(event, data) { (this.events[event] || []).forEach(cb => cb(data)); }
};
window.__MICRO_APP_EVENT_BUS__ = eventBus;
```

#### 真实面试题

**题目：微前端架构解决了什么问题？你了解哪些实现方案？**

**满分答案：**

**解决的问题：**
1. 大型应用拆分，独立开发部署
2. 多团队协作，减少代码冲突
3. 技术栈隔离，支持渐进式迁移

**三种主流方案：**
1. **qiankun**：JS 沙箱 + CSS 隔离，最成熟，适合大多数场景
2. **Module Federation**：Webpack 5 原生支持，共享依赖，性能最好，适合同技术栈
3. **iframe**：隔离最彻底，但性能差、通信复杂

**核心挑战：** 样式隔离（Shadow DOM / CSS Modules）、JS 沙箱（Proxy 代理 window）、应用间通信

## 14.x 技术规范推动与组件库建设

#### 知识点详解

**推动技术规范的步骤：**

```

   - 记录 Bug 类型分布
   - 测量构建/开发效率

2. 提出方案 → 小范围验证
   - 先在自己负责的模块试点
   - 收集数据证明效果

3. 推广 → 降低阻力
   - 写文档、做分享
   - 提供迁移工具
   - 渐进式而非强制
```

## 14.x 老旧项目技术重构与迁移

#### 知识点详解

**重构前的评估：**

```
评估维度：
1. 技术债务量化
   - 代码复杂度（圈复杂度）
   - 测试覆盖率
   - 依赖过期程度

2. 业务影响评估
   - 重构期间的功能稳定性
   - 用户影响范围

3. 收益预估
   - 开发效率提升
   - 性能改善
   - 维护成本降低
```

**渐进式重构策略（绞杀者模式）：**

```javascript
// 绞杀者模式（Strangler Fig Pattern）
// 新功能用新技术栈，老功能逐步迁移，不一次性重写

// 例：jQuery 项目迁移到 React
// 1. 新页面用 React 开发
// 2. 老页面中的独立模块逐步替换
// 3. 最终完全迁移

// 技术手段：在 jQuery 页面中挂载 React 组件
function mountReactComponent(containerId, Component, props) {
    const container = document.getElementById(containerId);
    if (container) {
        ReactDOM.render(<Component {...props} />, container);
    }
}

// 老页面中使用
mountReactComponent('new-feature-container', NewFeatureComponent, {
    data: window.legacyData
});
```

**依赖升级策略：**

```bash
# 1. 检查过期依赖
npm outdated

# 2. 自动升级（谨慎）
npx npm-check-updates -u

# 3. 分批升级（降低风险）
# 先升级开发依赖（ESLint、Prettier）
# 再升级工具链（Webpack、Babel）
# 最后升级核心框架（React、Vue）

# 4. 每次升级后运行测试
npm test
```

**重构检查清单：**

```
代码层面：
□ 消除重复代码（DRY 原则）
□ 拆分大文件（单文件 < 300 行）
□ 统一命名规范
□ 添加 TypeScript 类型

工程层面：
□ 升级构建工具（Webpack → Vite）
□ 添加单元测试
□ 配置 ESLint + Prettier
□ 优化 CI/CD 流程

性能层面：
□ 分析打包体积（Bundle Analyzer）
□ 代码分割（动态 import）
□ 图片优化
□ 缓存策略
```

**风险控制：**

```javascript
// 1. 功能开关（Feature Flag）
// 新旧代码并存，通过开关控制
const useNewFeature = featureFlags.get('new-checkout-flow');

if (useNewFeature) {
    return <NewCheckoutFlow />;
} else {
    return <OldCheckoutFlow />;
}

// 2. 灰度发布
// 先对 5% 用户开放，观察指标，再逐步放量

// 3. 回滚方案
// 保留旧版本代码，出问题立即切回
```

#### 真实面试题

**题目：面对一个老旧且复杂的项目，你会如何进行技术重构或迁移？**

**满分答案：**

**核心原则：渐进式，不一次性重写**

**四步走：**
1. **评估**：量化技术债务，评估业务影响，确定优先级
2. **策略**：绞杀者模式（新功能用新技术，老功能逐步替换）
3. **执行**：分批升级依赖 → 工程化改造 → 核心模块重构
4. **保障**：功能开关 + 灰度发布 + 完善的回滚方案

**关键经验：**
- 先加测试，再重构（没有测试的重构是盲目的）
- 每次只改一件事，保持可回滚
- 用数据验证重构效果（性能指标、错误率）
